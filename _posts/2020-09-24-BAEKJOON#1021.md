## [BAEKJOON #1021](https://www.acmicpc.net/problem/1021)

```python
n,m= map(int,input().split())
pick=list(map(int,input().split()))
number=list(range(1,n+1))

from collections import deque
cnt=0
_pick=deque(pick)
_number=deque(number)

while _pick:
    a=_pick[0]
    while _number[0]!=a:
        idx=_number.index(a)
        if idx <= len(_number)//2:
            _number.append(_number.popleft())
        else:
            _number.appendleft(_number.pop())
        cnt+=1
    _number.popleft()
    _pick.popleft()
print(cnt)
```

```python
n,m= map(int,input().split())

number=list(range(1,n+1))

cnt=0

while pick:
    a=pick[0]
    while number[0]!=a:
        idx=number.index(a)
        if idx <= len(number)//2:
            number.append(number.pop(0))
        else:
            number.insert(0,number.pop())
        cnt+=1
    number.pop(0)
    pick.pop(0)
print(cnt)
```
