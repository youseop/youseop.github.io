```python
import sys
read=sys.stdin.readline
from collections import deque

dx=[1,0,0,-1]
dy=[0,1,-1,0]

n=int(input())
height=list(list(map(int,read().split())) for _ in range(n))

#μ§€μ—­ μµλ€, μµμ† λ†’μ΄ μ €μ¥
max_h=0
min_h=sys.maxsize
for h in height:
    for x in h:
        max_h=max(max_h,x)
        min_h=min(min_h,x)

#hλ†’μ΄λ§νΌ λ¬Όμ΄ μ°Όμ„ λ•, μ•μ „μμ—­ κ°―μ νƒμƒ‰
def safe(h):
    global count
    #ν•΄λ‹Ή μ§€μ—­μ„ λ°©λ¬Έν–λ”μ§€ ν™•μΈ
    visit = list(list(True for _ in range(n)) for _ in range(n))
    for i in range(n):
        for j in range(n):
            #λ¬Όμ— μ κΈ°μ§€ μ•μ•κ³ , μµμ΄λ΅ λ°©λ¬Έν•λ” μ λ“¤μ— λ€ν•΄ νƒμƒ‰
            if visit[i][j] and height[i][j]>=h:
                #μμ—­μ΄ ν• κ³³ μ¶”κ°€ λμ—μμΌλ΅ 'count += 1'
                count+=1
                #BFS νƒμƒ‰μ„ ν†µν•΄ λ¬Όμ— μ κΈ°μ§€ μ•μ€ μΈμ ‘μμ—­μ„ λ¨λ‘ λ°©λ¬Έμ²λ¦¬
                point = deque([[i,j]])
                visit[i][j]=False
                while point:
                    a,b=point.popleft()
                    for x,y in zip(dx,dy):
                        ax,by=a+x,b+y
                        if 0<=ax<n and 0<=by<n and visit[ax][by] and height[ax][by]>=h:
                            visit[ax][by]=False
                            point.append([ax,by])

answer=1
for h in range(min_h,max_h+1):
    count=0 #hλ†’μ΄ λ―Έλ§μ μ§€μ—­λ“¤μ΄ μΉ¨μλμ—μ„ λ• μ•μ „μμ—­μ κ°μ
    safe(h)
    answer=max(answer,count)

print(answer)
```
