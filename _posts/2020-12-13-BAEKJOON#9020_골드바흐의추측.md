#include<iostream>

#include<algorithm>

int main() {
	int N;
	cin >> N;
	vector <int>num;

	for (int i = 0; i < N; i++) {
		int a;
		cin >> a;

		arr[i] = 1;
	}

	for (int i = 2; i < max;i++) {
		if (arr[i]) {
			for (int j = i * 2; j < max+1; j+=i) {
				arr[j] = 0;
			}
		}
	}

	for (int i = 0; i < N; i++) {
		for (int j = 0 ; j <= num[i]/2; j++) {
			if (arr[num[i]/2-j] & arr[num[i] / 2 + j]) {
				cout << num[i] / 2 - j <<" "<< num[i] / 2 + j <<"\n";
				break;
			}
		}
	}
}
